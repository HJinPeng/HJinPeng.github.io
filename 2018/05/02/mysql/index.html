MySql

发表于 2018-05-02

  1. 开启/关闭mysql
    net start mysql
    net stop mysql

  2. 进入/退出mysql
    mysql -uroot -pmysql
    exit

  3. 输入错误，回到上级
    \c

  4. 创建数据库
    CREATE DATABASE

  5. 修改数据库
    ALTER DATABASE t2 CHARACTER SET = utf8;

  6. 显示所有数据库
    SHOW DATABASES

  7. 显示某个数据库
    SHOW CREATE DATABASE t2

  8. 删除数据库
    DROP DATABASE t1;

  9. 使用数据库
    USE 数据库名

  10. 创建表

    CREATE TABLE tb1(
-> username VARCHAR(20),
-> age TINYINT UNSIGNED,
-> salary FLOAT(8,2) UNSIGNED
-> );

  11. 查看表
    SHOW TABLES;
    SHOW TABLES FROM mysql;

  12. 查看数据表结构
    SHOW COLUMNS FROM tb1;

  13. 记录的插入
    INSERT 表名 VALUES(全部数据)
    INSERT 表名(行名) VALUES(部分数据)
    INSERT users VALUES(NULL,'JOHN',123,2,1); NULL默认自动编号
    INSERT users VALUES(DEFAULT,'TOM',123,2,1); DEFAULT==NULL;
    INSERT users SET username='BEN',password='456';

  14. 记录的查询
    SELECT * FROM 表名

  15. 数据不能为空

     CREATE TABLE tb2(
-> username VARCHAR(20) NOT NULL,
-> age TINYINT UNSIGNED NULL
-> );

  16. 主键PRIMARY KEY

    1. 自动编号＋主键

      CREATE TABLE tb3(
-> id SMALLINT UNSIGNED AUTO_INCREMENT     PRIMARY KEY,
-> username VARCHAR(30) NOT NULL
-> );

    2. 主键（无自动编号，不可重复数据）

      CREATE TABLE tb4(
-> id SMALLINT UNSIGNED PRIMARY KEY,
-> username VARCHAR(30) NOT NULL
-> );

  17. 唯一约束UNIQUE KEY
    username VARCHAR(20) NOT NULL UNIQUE KEY,

  18. 默认值DEFAULT
    sex ENUM('1','2','3') DEFAULT '3'

  19. 外键约束 FOREIGN KEY(子表的列名) REFERENCES 父表 (父表的列名)

    CREATE TABLE provinces(
-> id SMALLINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
-> pname VARCHAR(20) NOT NULL
-> );

CREATE TABLE user(
-> id SMALLINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
-> username VARCHAR(10) NOT NULL,
-> pid SMALLINT UNSIGNED,
-> FOREIGN KEY (pid) REFERENCES provinces (id)
-> );

  20. 添加删除列
    ALTER TABLE users1 ADD password VARCHAR(30) NOT NULL AFTER username;
    ALTER TABLE users1 DROP truename;

  21. 添加约束
    主键约束：ALTER TABLE users2 ADD CONSTRAINT PK_users2_id PRIMARY KEY (id);
    唯一约束：ALTER TABLE users2 ADD UNIQUE (username);
    外键约束：ALTER TABLE users2 ADD FOREIGN KEY (pid) REFERENCES provinces (id);
    默认约束：ALTER TABLE users2 ALTER age SET DEFAULT 15;
    删除默认约束：ALTER TABLE users2 ALTER age DROP DEFAULT;
    删除唯一约束：

    查看约束名：`SHOW INDEXES FROM users2\G;`
删除索引：`ALTER TABLE users2 DROP INDEX username;`

    删除外键约束 ：

    查看外键约束名：`SHOW CREATE TABLE   users2;`
如CONSTRAINT `users2_ibfk_1`
删除：ALTER TABLE users2 DROP FOREIGN KEY users2_ibfk_1;

  22. 查看约束名字
    普通形式：SHOW INDEXES FROM users2;
    网格形式：SHOW INDEXES FROM users2\G;

  23. 修改列定义 MODIFY
    ALTER TABLE users2 MODIFY id TINYINT UNSIGNED NOT NULL FIRST;

  24. 修改列名称（同时修改列定义）
    ALTER TABLE users2 CHANGE pid p_id TINYINT UNSIGNED NOT NULL;

  25. 修改数据表的名称
    ALTER TABLE users2 RENAME users3;
    RENAME TABLE users3 TO users2;

  26. 单表更新
    UPDATE users SET age=age-id,sex=0;
    UPDATE users SET age=age+10 WHERE id%2=0;

  27. 单表删除
    DELETE FROM users WHERE id=6;
    下一个插入的数据id不会是删除的这个。

  28. 查询数据
    SELECT id,username FROM users; 先id再username
    SELECT username,id FROM users; 先username再id
    SELECT users.id,users.username FROM users;某个数据表的列名用.

  29. 别名AS
    SELECT id AS userID,username AS uname FROM users;改成AS后的名字

  30. GROUP BY 语句对查询结果分组
    SELECT sex,age FROM users GROUP BY sex,age;

    +------+-----+
| sex  | age |
+------+-----+
| NULL |  10 |
|    0 |   4 |
|    0 |   6 |
|    0 |  15 |
|    0 |  21 |
|    1 |   2 |
+------+-----+

    SELECT sex FROM users GROUP BY sex HAVING count(id)>3;

  31. 对查询结果进行排序[ASC|DESC]
    SELECT * FROM users ORDER BY age DESC,id DESC;

  32. 限制查询数量
    SELECT * FROM users LIMIT 2,2; 从0开始算的

  33. 添加数据到另一个表
    INSERT test(username) SELECT username FROM users WHERE age>=10;
